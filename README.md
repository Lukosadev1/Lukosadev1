        print("[4] Bölme (/)")
        print("[0] Çıkış")
        
        choice = input("\nSeçiminiz: ")
        
        if choice == "0":
            print("Görüşmek üzere! 👋")
            break
            
        num1 = float(input("İlk sayı: "))
        num2 = float(input("İkinci sayı: "))

            print(f"Sonuç: {num1} × {num2} = {num1 * num2}")
        elif choice == "4":

            else:
                print("❌ Hata: Sıfıra bölme yapılamaz!")
